// --- 3. MASOFANI HISOBLASH (1 KM = 2000 SO'M) ---
function getDistance(lat1, lon1, lat2, lon2) {
    const R = 6371;
    const dLat = (lat2 - lat1) * Math.PI / 180;
    const dLon = (lon2 - lon1) * Math.PI / 180;
    const a = Math.sin(dLat / 2) * Math.sin(dLat / 2) +
        Math.cos(lat1 * Math.PI / 180) * Math.cos(lat2 * Math.PI / 180) * Math.sin(dLon / 2) * Math.sin(dLon / 2);
    return R * 2 * Math.atan2(Math.sqrt(a), Math.sqrt(1 - a));
}

// --- 4. LOKATSIYA ANIQLASH ---
function initLocation() {
    const locText = document.getElementById("locationText");
    const locBtn = document.getElementById("locationBtn") || (locText ? locText.parentElement : null);

    if (navigator.geolocation && locText) {
        navigator.geolocation.getCurrentPosition(async (pos) => {
            const lat = pos.coords.latitude;
            const lon = pos.coords.longitude;
            userCoords = { lat, lon };

            try {
                const res = await fetch(`https://nominatim.openstreetmap.org/reverse?format=json&lat=${lat}&lon=${lon}&addressdetails=1`);
                const data = await res.json();

                if (data.address) {
                    const a = data.address;
                    const region = a.state || a.region || "Samarqand";
                    const area = a.suburb || a.neighbourhood || a.residential || "";
                    const point = data.name || a.amenity || a.shop || "";
                    
                    let parts = [];
                    if (region) parts.push(region);
                    if (area) parts.push(area);
                    if (point) parts.push(point);
                    if (!point && a.road) parts.push(a.road);

                    const coordString = `${lat.toFixed(6)}, ${lon.toFixed(6)}`;
                    userAddress = `${parts.join(", ")} (${coordString})`;
                    locText.textContent = userAddress;

                    if (locBtn) {
                        locBtn.onclick = () => window.open(`https://www.google.com/maps?q=${lat},${lon}`, '_blank');
                        locBtn.style.cursor = "pointer";
                    }
                }
            } catch (e) {
                locText.textContent = "Manzilni aniqlashda xatolik";
            }
            updateCartUI();
        }, (err) => {
            locText.textContent = "GPS ruxsat berilmadi";
            updateCartUI();
        }, { enableHighAccuracy: true });
    }
}

function openCart() {
    document.getElementById("cartPanel").classList.add("active");
    document.getElementById("cartOverlay").classList.add("active");
    renderCart();
}

function closeCartFn() {
    document.getElementById("cartPanel").classList.remove("active");
    document.getElementById("cartOverlay").classList.remove("active");
}

function renderCart() {
    const box = document.getElementById("cartItems");
    const totalText = document.getElementById("cartTotal");
    if (!box) return;

    box.innerHTML = "";
    let total = 0;

    Object.entries(cart).forEach(([name, item]) => {
        total += item.price * item.quantity;
        box.innerHTML += `
        <div class="cart-item">
            <img src="${item.img}" onerror="this.src='8.jpg'">
            <div class="cart-info">
                <h4>${name}</h4>
                <p>${Number(item.price).toLocaleString()} so'm</p>
                <div class="cart-actions">
                    <button onclick="changeQty('${name}', ${item.price}, -1)">−</button>
                    <span>${item.quantity}</span>
                    <button onclick="changeQty('${name}', ${item.price}, 1)">+</button>
                    <button class="cart-remove" onclick="removeItem('${name}')">🗑</button>
                </div>
            </div>
        </div>`;
    });

    if (totalText) totalText.textContent = total.toLocaleString() + " so'm";
    localStorage.setItem("cart", JSON.stringify(cart));
    updateCartUI();
}

function changeQty(name, price, delta, img = "8.jpg") {
    if (!cart[name] && delta > 0) {
        // Agar menyudan qo'shilayotgan bo'lsa, rasmni allFoods ichidan qidiramiz
        const foodData = allFoods.find(f => f.name === name);
        cart[name] = { price: price, quantity: 0, img: foodData ? foodData.img : img };
    }
    
    if (cart[name]) {
        cart[name].quantity += delta;
        if (cart[name].quantity <= 0) delete cart[name];
    }
    
    renderCart();
    renderFoods(activeCategory);
}

function removeItem(name) {
    delete cart[name];
    renderCart();
    renderFoods(activeCategory);
}

function updateCartUI() {
    let subtotal = 0;
    let count = 0;
    Object.values(cart).forEach(i => { 
        subtotal += i.price * i.quantity; 
        count += i.quantity; 
    });

    const panel = document.getElementById("bottomCartPanel");
    const nextBtn = document.getElementById("nextBtn");
    const deliveryLabel = document.getElementById("deliveryText");
    const totalPriceText = document.getElementById("totalPriceText");

    if (panel) {
        if (count > 0) {
            panel.style.display = "block";
            if (userCoords) {
                const dist = getDistance(RESTAURANT_COORDS.lat, RESTAURANT_COORDS.lon, userCoords.lat, userCoords.lon);
                const delivery = Math.round(dist * 2000);

                if (deliveryLabel) deliveryLabel.textContent = `Yetkazib berish: ${delivery.toLocaleString()} so'm (${dist.toFixed(1)} km)`;
                if (totalPriceText) totalPriceText.textContent = (subtotal + delivery).toLocaleString() + " so'm";
                if (nextBtn) { 
                    nextBtn.disabled = false; 
                    nextBtn.style.background = "#ffcc00"; 
                    nextBtn.style.opacity = "1"; 
                }
            } else {
                if (deliveryLabel) deliveryLabel.textContent = "Manzil aniqlanmoqda...";
                if (nextBtn) { nextBtn.disabled = true; nextBtn.style.background = "#cccccc"; }
            }
        } else {
            panel.style.display = "none";
        }
    }
}

function renderFoods(category) {
    const display = document.getElementById("foodDisplay");
    if (!display) return;
    display.innerHTML = "";
    
    const filtered = allFoods.filter(f => f.category === category);

    filtered.forEach(food => {
        const isOut = food.status === "out_of_stock";
        const count = cart[food.name]?.quantity || 0;

        display.innerHTML += `
            <div class="food-card ${isOut ? 'out-of-stock' : ''}">
                <img src="${food.img}" onerror="this.src='8.jpg'">
                <div class="food-info">
                    <h3>${food.name}</h3>
                    <p>${Number(food.price).toLocaleString()} so'm</p>
                    ${!isOut ? `
                    <div class="quantity-controls">
                        <button onclick="changeQty('${food.name}', -1)">-</button>
                        <span>${count}</span>
                        <button onclick="changeQty('${food.name}', 1)">+</button>
                    </div>` : '<div class="out-label">TUGAGAN</div>'}
                </div>
            </div>`;
    });
}

function closePage() { document.getElementById("secretPage").style.display = "none"; }

function searchFoodForEdit() {
    const q = document.getElementById("adminSearchInput").value.toLowerCase();
    editingIndex = allFoods.findIndex(f => f.name.toLowerCase().includes(q));
    if (editingIndex > -1) {
        const f = allFoods[editingIndex];
        document.getElementById("editFields").style.display = "block";
        document.getElementById("editName").value = f.name;
        document.getElementById("editPrice").value = f.price;
        document.getElementById("editImg").value = f.img;
        document.getElementById("editStatus").value = f.status;
    } else alert("Topilmadi!");
}

function saveEditedFood() {
    const status = document.getElementById("editStatus").value;
    if (status === "delete") {
        allFoods.splice(editingIndex, 1);
    } else {
        allFoods[editingIndex].name = document.getElementById("editName").value;
        allFoods[editingIndex].price = Number(document.getElementById("editPrice").value);
        allFoods[editingIndex].img = document.getElementById("editImg").value;
        allFoods[editingIndex].status = status;
    }
    localStorage.setItem("menuItems", JSON.stringify(allFoods));
    location.reload();
}
